    /* Enter your code here. Read input from STDIN. Print output to STDOUT */ 

   int fib(int n) 
{ 
    if (n <= 1) 
        return n; 
    return fib(n-1) + fib(n-2); 
} 
  
int main () 
{ 
    long n= 0;
    cin>>n;
    cout << fib(n);
    return 0;
}
