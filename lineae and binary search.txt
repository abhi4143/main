#include<iostream>
#include<stdlib.h>
#include<math.h>

using namespace std;
void linearSearch(int arr[],int n,int key)
{
	int i,flag=0;
	for(i=0;i<n;i++)
	{
		if(arr[i]==key)
		{
			flag=1;
			break;
		}
		else
		flag=0;
	}
	if(flag==1)
	cout<<"Element "<<key<<" found"<<endl;
	else
	cout<<"Element "<<key<<" not found"<<endl;
}
int BinarySearch(int arr[], int l, int r, int x) 
{ 
    if (r >= l) { 
        int mid = l + (r - l) / 2;  
        if (arr[mid] == x) 
            return mid; 
        if (arr[mid] > x) 
            return BinarySearch(arr, l, mid - 1, x); 
  
        return BinarySearch(arr, mid + 1, r, x); 
    }
    return -1; 
} 
int main()
{
	int arr[100],i,choice,key;
	for(i=0;i<30;i++)
	{

	}

	{
		case 1: cout<<"Enter the key element to be searched\n";
				cin>>key;

		case 2: cout<<"Enter the key element to be searched\n";
				cin>>key;
				int result;

				if(result==-1)

	}
}
