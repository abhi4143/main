#include<stdio.h>
void floyd(int [100][100],int);
int min(int,int);
int main()
{
	int e[100][100],i,j,n;
	printf("Enter number of vertices\n");
	scanf("%d",&n);
	printf("Enter the edges with weight\n");
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			scanf("%d",&e[i][j]);
			
		}
	}
	floyd(e,n);
	printf("Transitive closure is:\n");
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			printf("%d\t",e[i][j]);
		}
		printf("\n");
	}
	return 0;
}
void floyd(int e[100][100],int n)
{
	int i,j,k;
	for(k=1;k<=n;k++)
	{
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				e[i][j]=min((e[i][j]),(e[i][k]+e[k][j]));
			}
		}
	}
}
int min(int a,int b)
{
	if(a<b)
	return a;
	else
	return b;
}
